2020/12/24 画3D图

library(rgl)
x=c(0,0,1)
y=c(0,0.5,1)
z=c(0,1,1)
plot3d(x, y, z,
       type = "s", xlab = "", ylab = "", zlab = "",
       box = FALSE, axes = F, radius = c(0.3,0.45,0.3),
       col = c("pink","blue", "pink")
)
ind = c(2,1,2,3)
lines3d(x[ind], y[ind], z[ind], lwd = 2)
